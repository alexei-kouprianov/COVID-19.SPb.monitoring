################################################################
# Файл SPb.united.csv формируется автоматически на основе данных из трех файлов:
#
# (1) SPb.stopkoronavirus_archived.csv : архив данных по Петербургу с сайта "стопкоронавирус.рф" (обновляется ежедневно);
# (2) gov.spb.ru_archived.csv : сводный архив данных из ежедневных информационных бюллетеней Межведомственного городского совета по борьбе с распространением в Санкт-Петербурге новой коронавирусной инфекции (со 2 августа 2021 г., за исключением случаев последующего обнаружения данных за период до этой даты, не обновляется);
# (3) spb.gov.v.2021-08-03.csv : сводный архив данных из ежедневных обновлений страницы https://www.gov.spb.ru/covid-19/ (обновляется ежедневно);
# 
################################################################
# Описание переменных в файле SPb.united.csv
#
# За исключением особо оговоренных случаев ("TIME.sk" и "DATE.spb"), все переменные представлены целочисленными значениями. Отсутствующие данные обозначены кодом NA;
# 
# "TIME.sk" : дата и время публикации ежедневной сводки на сайте "стопкоронавирус.рф" ("YYYY-MM-DD hh:mm:ss");
# "CONFIRMED.sk" : количество новых подтвержденных случаев в ежедневной сводке на сайте "стопкоронавирус.рф";
# "RECOVERED.sk" : количество выздоровевших в ежедневной сводке на сайте "стопкоронавирус.рф";
# "DEATHS.sk" : количество смертей в ежедневной сводке на сайте "стопкоронавирус.рф";
# "ACTIVE.sk" : количество активных случаев согласно ежедневной сводке на сайте "стопкоронавирус.рф" (вычисляемый параметр);
# "DATE.spb" : дата публикации ежедневной сводки на сайте gov.spb.ru, с 3 августа 2021 г. -- на странице https://www.gov.spb.ru/covid-19/ ("YYYY-MM-DD");
# "CONFIRMED.spb" : количество новых подтвержденных случаев в ежедневной сводке на сайте gov.spb.ru;
# "RECOVERED.spb" : количество выздоровевших в ежедневной сводке на сайте gov.spb.ru;
# "ACTIVE.spb" : количество активных случаев (больные на амбулаторном лечении и в стационарах);
# "IN_HOSPITALS.spb" : количество больных, находящихся под наблюдением в стационарах;
# "HOSPITALIZED_TODAY.spb" : количество госпитализированных в течение дня;
# "PCR_TESTS" : количество PCR-тестов на COVID-19;
# "V1.CS" : накопленная сумма вакцинированных хотя бы одной дозой (без различения вакцин);
# "V1.TODAY" : количество вакцинированных первой дозой за день (без различения вакцин);
# "V2.CS" : накопленная сумма завершивших курс вакцинации (без различения вакцин);
# "V2.TODAY" : количество завершивших курс вакцинации (без различения вакцин);
# "V.REMAINING" : количество доступных доз вакцины (без различения вакцин);
# "DESIGNATED_BEDS" : количество коек, отведенных под COVID-19 в стационарах Петербурга;
# "VENT" : количество пациентов на ИВЛ;
# "VR1.CS" : накопленная сумма ревакцинированных хотя бы одной дозой (без различения вакцин);
# "VR1.TODAY" : количество ревакцинированных первой дозой за день (без различения вакцин);
# "VR2.CS" : накопленная сумма завершивших курс ревакцинации (без различения вакцин);
# "VR2.TODAY" : количество завершивших курс ревакцинации за день (без различения вакцин);
# "FREE_BEDS" : свободные койки (из числа отведенных под COVID-19 в стационарах Петербурга);
# "OCCUPIED_BEDS_CALCULATED" : расчетный показатель, количество занятых коек (из числа отведенных под COVID-19 в стационарах Петербурга) "OCCUPIED_BEDS_CALCULATED" = "DESIGNATED_BEDS" - "FREE_BEDS";
# "WEEKLY.REPORTS.BEDS_OCCUPIED.hard" : койки, занятые под тяжелые случаи (на конец недели, по еженедельным отчетам);
# "WEEKLY.REPORTS.BEDS_OCCUPIED.mid" : койки, занятые под случаи средней тяжести (на конец недели, по еженедельным отчетам);
# "WEEKLY.REPORTS.BEDS_OCCUPIED.lite" : койки, занятые под легкие случаи (на конец недели, по еженедельным отчетам);
# "WEEKLY.REPORTS.BEDS_FREE" : свободные койки (на конец недели, по еженедельным отчетам), не равно "FREE_BEDS";
# "WEEKDAY.num" : номер дня недели, начиная с понедельника;
# "WEEKDAY.ru" : день недели, сокращенно, на русском языке;
# "WEEKDAY.en" : день недели, сокращенно, на английском языке;
#
################################################################
# Источники данных:
#
# "TIME.sk", "CONFIRMED.sk", "RECOVERED.sk", "DEATHS.sk", "ACTIVE.sk" : сайт "стопкоронавирус.рф", "ACTIVE.sk" после 9 июля 2021 г. перестали публковать на сайте;
# "DATE.spb" : добавлена автоматически;
# "CONFIRMED.spb", "RECOVERED.spb", "ACTIVE.spb" : ежедневные сводки на сайте gov.spb.ru;
# "IN_HOSPITALS.spb", "HOSPITALIZED_TODAY.spb" : ежедневные сводки на сайте gov.spb.ru, с 3 августа 2021 г. -- на странице https://www.gov.spb.ru/covid-19/;
# "PCR_TESTS" : ежедневные сводки в телеграм-канале "Петербург против коронавируса", с 3 августа 2021 г. -- на странице https://www.gov.spb.ru/covid-19/;
# "V1.CS", "V1.TODAY", "V2.CS", "V2.TODAY", "V.REMAINING" : ежедневные сводки на сайте gov.spb.ru, с 3 августа 2021 г. -- на странице https://www.gov.spb.ru/covid-19/;
# "DESIGNATED_BEDS", "VENT", "VR1.CS", "VR1.TODAY", "VR2.CS", "VR2.TODAY", "FREE_BEDS" : ежедневные сводки на сайте gov.spb.ru и сообщения СМИ, с 3 августа 2021 г. -- на странице https://www.gov.spb.ru/covid-19/;
# "OCCUPIED_BEDS_CALCULATED" : вычисляемый показатель. Рассчитывается на основе переменных "DESIGNATED_BEDS" и "FREE_BEDS" ("DESIGNATED_BEDS" - "FREE_BEDS" = "OCCUPIED_BEDS_CALCULATED"); 
# "WEEKLY.REPORTS.BEDS_OCCUPIED.hard", "WEEKLY.REPORTS.BEDS_OCCUPIED.mid", "WEEKLY.REPORTS.BEDS_OCCUPIED.lite", "WEEKLY.REPORTS.BEDS_FREE" : еженедельные отчеты на сайте gov.spb.ru (публиковались до 2 июня 2021 г.); 
# "WEEKDAY.num", "WEEKDAY.ru", "WEEKDAY.en" -- добавлены автоматически;
